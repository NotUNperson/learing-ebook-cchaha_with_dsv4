## 正文

### 练习目标

这次我们要搭建一个"前后端分离"的经典架构。场景设定：

一个简单的"留言板"应用。前端是纯静态 HTML 页面（由 Nginx 直接提供），后端是一个 Node.js 服务（Nginx 反向代理过去）。前端通过 JavaScript 的 `fetch` 调用后端 API，API 路径以 `/api/` 开头。

整个架构如下：

```
浏览器
  |
  | http://messageboard.local/
  v
Nginx (80 端口)
  |
  +-- /            --> 返回静态 HTML (html/frontend/index.html)
  +-- /api/messages --> 反向代理到 http://localhost:3000/messages
  +-- /api/add     --> 反向代理到 http://localhost:3000/add
```

### 步骤一：准备后端服务

创建 `backend/server.js`：

```javascript
const http = require('http');

const PORT = 3000;
const messages = [];

const server = http.createServer((req, res) => {
    const url = new URL(req.url, `http://${req.headers.host}`);

    // CORS 头（如果前端和后端不同域）
    res.setHeader('Access-Control-Allow-Origin', '*');
    res.setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
    res.setHeader('Access-Control-Allow-Headers', 'Content-Type');

    if (req.method === 'OPTIONS') {
        res.writeHead(204);
        res.end();
        return;
    }

    // GET /messages —— 获取留言列表
    if (req.method === 'GET' && url.pathname === '/messages') {
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({
            success: true,
            total: messages.length,
            messages: messages,
            clientIp: req.headers['x-real-ip'] || req.socket.remoteAddress,
            forwardedFor: req.headers['x-forwarded-for'] || '无'
        }));
        return;
    }

    // POST /add —— 添加留言
    if (req.method === 'POST' && url.pathname === '/add') {
        let body = '';
        req.on('data', chunk => body += chunk);
        req.on('end', () => {
            try {
                const data = JSON.parse(body);
                const msg = {
                    id: Date.now(),
                    name: data.name || '匿名',
                    content: data.content || '',
                    time: new Date().toISOString()
                };
                messages.unshift(msg);
                res.writeHead(200, { 'Content-Type': 'application/json' });
                res.end(JSON.stringify({
                    success: true,
                    message: msg,
                    clientIp: req.headers['x-real-ip'] || req.socket.remoteAddress
                }));
            } catch (e) {
                res.writeHead(400, { 'Content-Type': 'application/json' });
                res.end(JSON.stringify({ success: false, error: '数据格式错误' }));
            }
        });
        return;
    }

    // GET /slow —— 模拟慢请求（用于测试超时）
    if (req.method === 'GET' && url.pathname === '/slow') {
        console.log('收到慢请求，将在 15 秒后响应...');
        setTimeout(() => {
            res.writeHead(200, { 'Content-Type': 'application/json' });
            res.end(JSON.stringify({ success: true, message: '这个响应延迟了 15 秒' }));
        }, 15000);
        return;
    }

    // 404
    res.writeHead(404, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify({ success: false, error: '接口不存在' }));
});

server.listen(PORT, () => {
    console.log(`留言板后端已启动：http://localhost:${PORT}`);
    console.log('接口列表：');
    console.log('  GET  /messages  - 获取留言列表');
    console.log('  POST /add       - 添加留言');
    console.log('  GET  /slow      - 慢请求（15秒延迟，测试超时）');
});
```

### 步骤二：准备前端页面

创建 `frontend/index.html`：

```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>留言板</title>
    <style>
        body { font-family: "Microsoft YaHei",sans-serif; max-width:600px; margin:40px auto; padding:20px; }
        h1 { color:#2c3e50; border-bottom:3px solid #3498db; padding-bottom:10px; }
        .form { background:#f8f9fa; padding:20px; border-radius:8px; margin-bottom:30px; }
        .form input,.form textarea { width:100%; margin:8px 0; padding:10px; border:1px solid #ddd; border-radius:4px; font-size:14px; }
        .form textarea { height:80px; resize:vertical; }
        .form button { background:#3498db; color:white; border:none; padding:10px 24px; border-radius:4px; cursor:pointer; font-size:14px; }
        .message { background:white; padding:16px; margin:12px 0; border-radius:8px; box-shadow:0 2px 8px rgba(0,0,0,0.06); }
        .message .name { font-weight:bold; color:#3498db; }
        .message .time { color:#999; font-size:12px; float:right; }
        .message .content { margin-top:8px; color:#555; line-height:1.6; }
    </style>
</head>
<body>
    <h1>留言板</h1>
    <div class="form">
        <input type="text" id="nameInput" placeholder="你的昵称">
        <textarea id="contentInput" placeholder="说点什么..."></textarea>
        <button onclick="addMessage()">发表留言</button>
    </div>
    <div id="messageList"><p style="color:#999;">加载中...</p></div>

    <script>
        async function loadMessages() {
            try {
                const res = await fetch('/api/messages');
                const data = await res.json();
                const list = document.getElementById('messageList');
                if (data.messages.length === 0) {
                    list.innerHTML = '<p style="color:#999;text-align:center;">还没有留言，来发第一条吧！</p>';
                    return;
                }
                list.innerHTML = data.messages.map(m => `
                    <div class="message">
                        <span class="name">${escapeHtml(m.name)}</span>
                        <span class="time">${new Date(m.time).toLocaleString()}</span>
                        <div class="content">${escapeHtml(m.content)}</div>
                    </div>
                `).join('');
            } catch (e) {
                document.getElementById('messageList').innerHTML = '<p style="color:red;">加载失败：' + e.message + '</p>';
            }
        }
        async function addMessage() {
            const name = document.getElementById('nameInput').value.trim();
            const content = document.getElementById('contentInput').value.trim();
            if (!content) { alert('请输入留言内容'); return; }
            const res = await fetch('/api/add', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ name, content })
            });
            if (res.ok) {
                document.getElementById('contentInput').value = '';
                loadMessages();
            }
        }
        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }
        loadMessages();
    </script>
</body>
</html>
```

### 步骤四：验证要点

完成搭建后，逐一确认：

1. **访问前端页面**：`http://messageboard.local/` 能看到留言板界面
2. **发表留言**：填写昵称和内容，点击发表，留言能正常显示
3. **再次加载**：刷新页面，之前的留言还在（后端内存保存）
4. **查看客户端 IP**：用 `curl http://messageboard.local/api/messages` 查看后端返回的 `clientIp` 字段
5. **测试超时**：访问 `http://messageboard.local/api/slow`（如果 `proxy_read_timeout` 小于 15 秒，会看到 504 错误）
6. **查看日志**：检查 `access.log` 和 `error.log` 确认请求记录和错误信息正确

### 步骤五：额外挑战

- 把 `proxy_read_timeout` 调成 20 秒，再次访问 `/api/slow`，体验超时设置的调整效果
- 尝试修改 `X-Forwarded-For` 的值为自定义字符串，看看后端返回的 `forwardedFor` 字段变化
- 在 hosts 中尝试用不同域名访问，观察后端返回的 Host 头变化

## 动手试试

按照上面五个步骤，完整搭建这个留言板应用。整个过程涵盖了我们学到的所有反向代理核心知识——proxy_pass 路径替换、请求头设置、超时配置、静态文件 + 代理的混合配置。整个练习预计 15-20 分钟完成。

## 本节小结

一个完整的反向代理架构将静态文件服务（前端）和 API 代理（后端）统一在同一个域名下。前端通过相对路径调用 API，Nginx 在中间负责路由分发、请求头传递和超时保护。
